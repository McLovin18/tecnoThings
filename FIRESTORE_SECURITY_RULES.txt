rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // ========== FUNCIONES AUXILIARES ==========
    
    // Obtener rol del usuario actual
    function getUserRole() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role;
    }
    
    // Verificar si el usuario es admin
    function isAdmin() {
      return getUserRole() == 'admin';
    }
    
    // Verificar si el usuario es cliente
    function isClient() {
      return getUserRole() == 'cliente';
    }
    
    // Verificar si el usuario es propietario del documento
    function isOwner(uid) {
      return request.auth.uid == uid;
    }
    
    // Verificar autenticaci칩n
    function isAuthenticated() {
      return request.auth != null;
    }
    
    // ========== COLECCI칍N: USERS ==========
    match /users/{userId} {
      // Cada usuario puede leer su propio documento
      allow read: if isAuthenticated() && (isOwner(userId) || isAdmin());
      
      // Solo admins pueden crear/actualizar users
      allow create: if isAdmin();
      allow update: if isAdmin() || isOwner(userId);
      
      // Solo admins pueden eliminar
      allow delete: if isAdmin();
    }
    
    // ========== COLECCI칍N: PRODUCTS ==========
    match /products/{productId} {
      // Lectura p칰blica de productos activos (landing y visitantes)
      // Admins leen todo
      allow read: if resource.data.isActive == true || (isAuthenticated() && isAdmin());
      
      // Solo admins pueden crear productos
      allow create: if isAdmin();
      
      // Solo admins pueden actualizar
      allow update: if isAdmin();
      
      // Solo admins pueden eliminar
      allow delete: if isAdmin();
    }
    
    // ========== COLECCI칍N: ORDERS ==========
    match /orders/{orderId} {
      // Clientes pueden leer solo sus propias 칩rdenes
      // Admins pueden leer todas
      allow read: if isAuthenticated() && (
        (isClient() && isOwner(resource.data.userId)) ||
        isAdmin()
      );
      
      // Clientes pueden crear 칩rdenes
      allow create: if isAuthenticated() && isClient() && 
        request.resource.data.userId == request.auth.uid;
      
      // Solo admins pueden actualizar estado
      allow update: if isAdmin();
      
      // Solo admins pueden eliminar
      allow delete: if isAdmin();
    }
    
    // ========== COLECCI칍N: CARTS ==========
    match /carts/{userId} {
      // Cada usuario solo puede leer/escribir su carrito
      allow read: if isOwner(userId);
      allow write: if isOwner(userId);
    }
    
    // ========== COLECCI칍N: BLOGS ==========
    match /blogs/{blogId} {
      // Lectura p칰blica de blogs publicados
      // Admins ven todos
      allow read: if resource.data.isPublished == true || (isAuthenticated() && isAdmin());
      
      // Solo admins pueden crear
      allow create: if isAdmin();
      
      // Solo admins pueden actualizar
      allow update: if isAdmin();
      
      // Solo admins pueden eliminar
      allow delete: if isAdmin();
    }
    
    // ========== COLECCI칍N: LANDING PAGE ==========
    match /landingPage/{document=**} {
      // Todos pueden leer la landing
      allow read: if true;
      
      // Solo admins pueden escribir
      allow write: if isAdmin();
    }
    
    // ========== COLECCI칍N: USER PREFERENCES ==========
    match /user_preferences/{userId} {
      // Cada usuario solo accede a sus preferencias
      allow read, write: if isOwner(userId);
    }
